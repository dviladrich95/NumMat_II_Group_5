\newcommand{\addFramedFig}[5]
{
	\begin{figure}[#1]
		\begin{center}
			\tcbox[sharp corners,
				   boxsep=1pt,
				   boxrule=1pt, 
				   colframe=black,
				   colback=white]
				   {\includegraphics[width=#2, keepaspectratio]{#3.#4}}
			\caption{#5}
			\label{fig:#3}
		\end{center}
	\end{figure}
}

\newcommand{\addTable}[3]
{

}
